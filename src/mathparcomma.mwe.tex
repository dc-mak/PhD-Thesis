 {

  \int_zero:N \l_tmpa_int
  \seq_map_inline:Nn \l_tmpa_seq
   {
     \int_incr:N \l_tmpa_int
     \ensuremath{##1}%
     \int_compare:nNnTF { \l_tmpa_int } < { \seq_count:N \l_tmpa_seq }
       { ,\and }
       { }
   }
  \end{mathpar}
 }
 
\ExplSyntaxOff
